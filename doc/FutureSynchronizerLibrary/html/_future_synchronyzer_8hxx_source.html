103  void launch()
104  {
105  FunctionPtr f;
106  //Initialize parts
107  size_t size = m_numThreads < m_functions.size() ?
108  m_numThreads : m_functions.size();
109  for (size_t i = 0; i < size; ++i)
110  {
111  f = getNextFunction();
112  std::future<OutputResult> future =
113  FutureFactory::template createFuture<OutputResult>(*f);
114  m_futures.push_back(std::move(future));
115  }
116 
117  //While all functions are not launched;
118  while(!m_functions.empty())
119  {
120  for (size_t numThread = 0; numThread < m_numThreads; ++numThread)
121  {
122  std::future_status status =
123  m_futures[numThread].wait_for(std::chrono::microseconds(0));
124  //If the current future is ready
125  if (status == std::future_status::ready)
126  {
127  //Then, we calculate the result (if necessary)
128  applyResult(m_futures[numThread]);
129  //And we create a new future.
130  f = getNextFunction();
131  std::future<OutputResult> future =
132  FutureFactory::template createFuture<OutputResult>(*f);
133  m_futures[numThread] = std::move(future);
134  if (m_functions.empty())
135  {
136  break;
137  }
138  }
139  }
140  }
141  }
